import { Field, Switch } from "@aortl/admin-react";

A toggle for on/off state. Use a switch when the change applies immediately (e.g. enable a feature). For form-submission booleans, prefer a [checkbox](../checkboxes/).

## Examples

### Default

:::example

```html
<button type="button" role="switch" aria-checked="false" class="switch" data-unchecked>
  <span class="switch-thumb"></span>
</button>
<button type="button" role="switch" aria-checked="true" class="switch" data-checked>
  <span class="switch-thumb"></span>
</button>
```

```tsx
<Switch />
<Switch defaultChecked />
```

:::

### Disabled

:::example

```html
<button type="button" role="switch" aria-checked="false" class="switch" data-unchecked disabled>
  <span class="switch-thumb"></span>
</button>
<button type="button" role="switch" aria-checked="true" class="switch" data-checked disabled>
  <span class="switch-thumb"></span>
</button>
```

```tsx
<Switch disabled />
<Switch disabled defaultChecked />
```

:::

### Inside a Field

:::example

```html
<div class="field" style="flex-direction: row; align-items: center; gap: 0.75rem;">
  <button type="button" role="switch" aria-checked="true" class="switch" data-checked>
    <span class="switch-thumb"></span>
  </button>
  <label class="field-label">Email notifications</label>
</div>
```

```tsx
<Field name="notify" style={{ flexDirection: "row", alignItems: "center", gap: "0.75rem" }}>
  <Switch defaultChecked />
  <Field.Label>Email notifications</Field.Label>
</Field>
```

:::
