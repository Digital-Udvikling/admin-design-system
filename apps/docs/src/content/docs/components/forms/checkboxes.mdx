import { Checkbox } from "@aortl/admin-react";
import Example from "../../../../components/Example.astro";
import { CheckboxFieldDemo } from "../../../../demos/CheckboxFieldDemo";

The React component wraps Base UI's `Checkbox.Root` (a `<button role="checkbox">`) so the indicator can be styled freely. The vanilla example styles a native `<input type="checkbox">` with the same class names — visually equivalent, different rendered DOM.

## Examples

### Default

<Example
  html={`<input type="checkbox" class="checkbox">
<input type="checkbox" class="checkbox" checked>`}
  react={`<Checkbox />
<Checkbox defaultChecked />`}
>
  <div style={{ display: "flex", gap: "0.75rem", alignItems: "center" }}>
    <Checkbox />
    <Checkbox defaultChecked />
  </div>
</Example>

### Disabled

<Example
  html={`<input type="checkbox" class="checkbox" disabled>
<input type="checkbox" class="checkbox" disabled checked>`}
  react={`<Checkbox disabled />
<Checkbox disabled defaultChecked />`}
>
  <div style={{ display: "flex", gap: "0.75rem", alignItems: "center" }}>

### Indeterminate

The indeterminate state is set imperatively in React; vanilla mirrors it by setting `indeterminate` on the DOM node. The CSS treats indeterminate identically to checked.

<Example
  html={`<input type="checkbox" class="checkbox" id="indeterm">
<script>document.getElementById("indeterm").indeterminate = true;</script>`}
  react={`<Checkbox indeterminate />`}
>
  <Checkbox indeterminate />
</Example>
